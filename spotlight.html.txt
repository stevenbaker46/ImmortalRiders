    <section class="spotlight-card">
      <div>
        <img id="photo" class="spotlight-photo" alt="Spotlight member photo" />
      </div>
      <div>
        <h2 id="name" style="margin:0;"></h2>
        <div class="spotlight-meta" id="meta"></div>

        <p class="spotlight-why" id="why"></p>

        <h3 style="margin: 18px 0 8px;">Highlights</h3>
        <ul class="spotlight-list" id="highlights"></ul>

        <blockquote id="quote"></blockquote>
      </div>
    </section>
  </main>

  <script>
    async function loadSpotlight() {
      const res = await fetch("data/spotlight.json", { cache: "no-store" });
      const data = await res.json();

      document.getElementById("month").textContent = data.month || "";
      document.getElementById("name").textContent = data.member?.name || "Member";
      document.getElementById("meta").textContent =
        [data.member?.chapter ? `Chapter: ${data.member.chapter}` : ""].filter(Boolean).join(" • ");

      const photo = document.getElementById("photo");
      photo.src = data.member?.photo || "assets/hero-fallback.jpg";

      document.getElementById("why").textContent = data.member?.why || "";

      const ul = document.getElementById("highlights");
      ul.innerHTML = "";
      (data.member?.highlights || []).forEach(item => {
        const li = document.createElement("li");
        li.textContent = item;
        ul.appendChild(li);
      });

      const quote = data.member?.quote ? `“${data.member.quote}”` : "";
      document.getElementById("quote").textContent = quote;
      document.getElementById("quote").style.display = quote ? "block" : "none";
    }

    loadSpotlight().catch(err => {
      console.error(err);
      document.getElementById("month").textContent = "Could not load spotlight.";
    });
  </script>
